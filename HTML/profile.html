// 快速示例：控制标题发光（仅展示）
.neon-title {
  background: linear-gradient(90deg, #00e6ff, #ff44d0);
  -webkit-background-clip: text;
  color: transparent;
  text-shadow: 0 0 12px rgba(0,230,255,0.6);
  animation: neon-flicker 3s infinite;
}
